# -*- coding: utf-8 -*-
import c4d
from c4d import plugins, documents, bitmaps
import threading

PLUGIN_ID = 1070002  # Message plugin ID (also used as SpecialEvent id)


class HttpServicePlugin(plugins.CommandData):
    def Execute(self, doc):
        print("start")
        return True


if __name__ == "__main__":
    plugins.RegisterCommandPlugin(
        id=PLUGIN_ID,
        str="http",
        info=0,
        icon=None,
        help="http插件",
        dat=HttpServicePlugin(),
    )
